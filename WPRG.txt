<?php
function isPrime($cyfra) {
    if ($cyfra <= 1) {
        return false;
    }
    for ($i = 2; $i <= sqrt($cyfra); $i++) {
        if ($cyfra % $i === 0) {
            return false;
        }
    }
    return true;
}

echo "Cyfry pierwsze z zakresu od 1 do 27:<br>";
for ($i = 1; $i <= 27; $i++) {
    if (isPrime($i)) {
        echo $i . "<br>";
    }
}
?>

<?php
function fibonacci($n) {
    $fibo = array();
    $fibo[0] = 0;
    $fibo[1] = 1;
    for ($i = 2; $i < $n; $i++) {
        $fibo[$i] = $fibo[$i - 1] + $fibo[$i - 2];
    }
    return $fibo;
}

$N = 27; // N-ta liczba Fibonacciego
$fibonacio = fibonacci($N);

echo "Nieparzyste elementy ciÄ…gu Fibonacciego od 1 do 27:<br>";
foreach ($fibonacio as $numer => $cyfra) {
    if ($cyfra % 2 !== 0) {
        echo ($numer + 1) . ": $cyfra<br>";
    }
}
?>
